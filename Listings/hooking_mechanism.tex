  "ClassName", 
  lpparam.classLoader, 
  "methodName", 
  XC_MethodHook() {
    @Throws(Throwable::class)
    override protected void beforeHookedMethod(
      MethodHookParam param) {
        super.beforeHookedMethod(param)
    }

    @Throws(Throwable::class)
    override protected void afterHookedMethod(
      MethodHookParam param) {
        super.afterHookedMethod(param)
    }
  }
